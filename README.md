# To delete a local branch
## git branch -d dev
## git branch -d test

# To delete a remote branch
## git push origin :dev
## git push origin :test

# To list all tags
## git tag

# To delete remote tag
## git push origin --delete v1.7

# To delete local tags
## git tag -d v1.7
